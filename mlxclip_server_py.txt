import argparse
import base64
import logging
import tempfile
import os
from io import BytesIO
import gc
from contextlib import asynccontextmanager

import asyncio
import mlx.core as mlx_core
import uvicorn
from fastapi import FastAPI, HTTPException, Body
from PIL import Image
from mlx_clip import mlx_clip

parser = argparse.ArgumentParser(description="MLX-Clip FastAPI embedding service")
parser.add_argument("--model_dir", required=True, help="Path to MLX CLIP model directory")
parser.add_argument("--host", default="localhost", help="The host the service will listen on.")
parser.add_argument("--port", type=int, default=5000, help="The port the service will listen on.")
parser.add_argument("--max-workers", type=int, default=16, help="The maximum number of concurrent mlx processes.")
_args = parser.parse_args()

logging.basicConfig(level=logging.INFO)
log = logging.getLogger(__name__)

log.info(f"Loading MLX-CLIP model from {_args.model_dir}")
clip = mlx_clip(_args.model_dir)

MODEL_NAME = "mlxclip#" + clip.hf_repo

app = FastAPI(title="MLX-Clip Service")

# START OF code to manage MLX queues

stream_pool = asyncio.Queue()

@app.on_event("startup")
async def setup_streams():
    dev = mlx_core.Device(mlx_core.gpu, 0)
    for _ in range(_args.max_workers):
        stream = mlx_core.new_stream(dev)
        await stream_pool.put(stream)

@asynccontextmanager
async def get_worker_stream():
    stream = await stream_pool.get()
    try:
        with mlx_core.stream(stream):
            yield stream
    finally:
        stream_pool.put_nowait(stream)

# END OF code to manage MLX queues

@app.post("/embeddings")
async def embeddings_text(payload: dict = Body(...)):

    if "content" not in payload:
        raise HTTPException(status_code=400, detail="Missing 'content'")

    async with get_worker_stream():
        
        try:
            embedding = clip.text_encoder(payload["content"])
            return {"embeddings": embedding, "model": MODEL_NAME}
        except Exception as e:
            log.exception("Text encoding failed")
            raise HTTPException(status_code=500, detail=str(e))
        finally:
            
            mlx_core.metal.clear_cache()
            gc.collect()

@app.post("/embeddings/image")
async def embeddings_image(payload: dict = Body(...)):

    async with get_worker_stream():    
        
        try:
            img_b64 = payload["image_data"][0]["data"]
            img_data = base64.b64decode(img_b64)
            img = Image.open(BytesIO(img_data)).convert("RGB")
        except Exception as e:
            log.exception("Image decoding failed")
            raise HTTPException(status_code=400, detail="Invalid image data")
        
        temp_path = None
        
        try:
            with tempfile.NamedTemporaryFile(delete=False, suffix=".png") as wr:
                img.save(wr.name, format="PNG")
                temp_path = wr.name

            embedding = clip.image_encoder(temp_path)
            return {"embeddings": embedding, "model": MODEL_NAME}

        except Exception as e:
            log.exception("Image encoding failed")
            raise HTTPException(status_code=500, detail=str(e))
        finally:
        
            if temp_path and os.path.exists(temp_path):
                os.remove(temp_path)
                
            mlx_core.metal.clear_cache()
            gc.collect()
        
if __name__ == "__main__":
    uvicorn.run(app, host=_args.host, port=_args.port)
