import tempfile
import base64
import os

from flask import Flask, request, jsonify

from transformers import AutoProcessor, AutoModel
import torch
from PIL import Image

# START OF something something something command line options
# https://github.com/sfomuseum/go-embeddings/issues/12

# 768 dimensions
# model_name = "google/siglip-base-patch16-224"
# model_name = "google/siglip2-base-patch16-224"

# 1024 dimensions
# model_name="google/siglip-large-patch16-256"
model_name="google/siglip2-large-patch16-256"

# END OF something something something command line options

processor = AutoProcessor.from_pretrained(model_name)
model     = AutoModel.from_pretrained(model_name).eval()

device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
model   = model.to(device)

app = Flask(__name__)

@app.route("/embeddings", methods=['POST'])
def embeddings():

    req = request.json

    inputs = processor(text=req["content"], return_tensors="pt").to(device)
    
    with torch.no_grad():
        vec = model.get_text_features(**inputs)
        vec = vec.squeeze(0) 
        
    vec = torch.nn.functional.normalize(vec, p=2, dim=0)
    rsp =  vec.cpu().numpy()

    embeddings = rsp.tolist()
    return jsonify({"embedding": embeddings, "model": model_name})
    
    
@app.route("/embeddings/image", methods=['POST'])
def embeddings_image():

    req = request.json
    body = base64.b64decode(req["image_data"][0]["data"])

    with tempfile.NamedTemporaryFile(delete_on_close=False, mode="wb") as wr:

        wr.write(body)
        wr.close()

        img = Image.open(wr.name).convert("RGB")
        os.remove(wr.name)
        
        inputs = processor(images=img, return_tensors="pt").to(device)

        with torch.no_grad():
            vec = model.get_image_features(**inputs)
            vec = vec.squeeze(0) 

        vec = torch.nn.functional.normalize(vec, p=2, dim=0)
        rsp = vec.cpu().numpy()
        
        embeddings = rsp.tolist()
        return jsonify({"embedding": embeddings, "model": model_name})
